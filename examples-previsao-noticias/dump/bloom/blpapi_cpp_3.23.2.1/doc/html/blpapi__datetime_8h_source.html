1217 {
1218  if ((year <= 0) || (year > 9999) || (month <= 0) || (month > 12)
1219  || (day <= 0) || (day > 31)) {
1220  return false;
1221  }
1222  if (year == 1752) {
1223  if (month == 9 && day > 2 && day < 14) {
1224  return false;
1225  }
1226  }
1227  if (day < 29) {
1228  return true;
1229  }
1230  switch (month) {
1231  case 1:
1232  case 3:
1233  case 5:
1234  case 7:
1235  case 8:
1236  case 10:
1237  case 12:
1238  return true;
1239 
1240  case 4:
1241  case 6:
1242  case 9:
1243  case 11: {
1244  if (day > 30) {
1245  return false;
1246  } else {
1247  return true;
1248  }
1249  }
1250  case 2: {
1251  if (isLeapYear(year)) {
1252  if (day > 29) {
1253  return false;
1254  } else {
1255  return true;
1256  }
1257  } else if (day > 28) {
